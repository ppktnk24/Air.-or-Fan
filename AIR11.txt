import cv2
import torch
from PIL import Image
from transformers import CLIPProcessor, CLIPModel

def main():
    print("Downloading/Loading the AI model (this may take a minute on the first run)...")
    model_id = "openai/clip-vit-base-patch32"
    processor = CLIPProcessor.from_pretrained(model_id)
    model = CLIPModel.from_pretrained(model_id)

    # Define the categories. We add a "background" category so it doesn't 
    # force a guess if neither object is in the frame.
    labels =["a fan", "an air conditioner", "a generic room background or person"]

    # Open the default camera (0 is usually the built-in webcam)
    cap = cv2.VideoCapture(0)
    
    if not cap.isOpened():
        print("Error: Could not open the webcam.")
        return

    print("Camera started! Show a fan or an AC to the camera. Press 'q' to quit.")

    while True:
        # Capture frame-by-frame
        ret, frame = cap.read()
        if not ret:
            break

        # OpenCV uses BGR colors, but the AI model expects RGB
        rgb_frame = cv2.cvtColor(frame, cv2.COLOR_BGR2RGB)
        pil_img = Image.fromarray(rgb_frame)

        # Prepare the image and text labels for the AI
        inputs = processor(text=labels, images=pil_img, return_tensors="pt", padding=True)
        
        # Pass the data through the AI model
        with torch.no_grad():
            outputs = model(**inputs)
        
        # Calculate probabilities (which label is the most likely)
        logits_per_image = outputs.logits_per_image
        probs = logits_per_image.softmax(dim=1).cpu().numpy()[0]
        
        # Find the highest probability
        max_idx = probs.argmax()
        predicted_label = labels[max_idx]
        confidence = probs[max_idx] * 100

        # Determine what text to display on the screen
        if "background" in predicted_label or confidence < 55:
            display_text = "Scanning for Fan or AC..."
            box_color = (0, 0, 255) # Red for scanning
        else:
            display_text = f"Detected: {predicted_label.title()} ({confidence:.1f}%)"
            box_color = (0, 255, 0) # Green for found

        # Draw the text on the video frame
        cv2.putText(frame, display_text, (10, 40), cv2.FONT_HERSHEY_SIMPLEX, 0.8, box_color, 2)
        
        # Show the video feed
        cv2.imshow("Fan vs AC Detector", frame)

        # Break the loop if the user presses 'q'
        if cv2.waitKey(1) & 0xFF == ord('q'):
            break

    # Release the camera and close the window
    cap.release()
    cv2.destroyAllWindows()

if __name__ == "__main__":
    main()
